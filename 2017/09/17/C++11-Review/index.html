C++11

类型推导

  • auto
    • 当不声明为指针或者引用时，auto的推导结果和初始化表达式抛弃引用和cv限定符后类型一致
    • 当声明为指针或者引用时，auto的推导结果将保持初始化表达式的cv属性
      (cv-qualifier,const和volatile限定符的统称)
    • auto不能用于函数参数
    • auto不能用于非静态成员变量
    • auto：需要初始化来确定类型
  • decltype
    • decltype(exp)
    • exp是标识符、类访问表达式、decltype(exp)和exp的类型一致
      class Foo
{
public:
  static const int Number = 0;
  int x;
};
const Foo foo = Foo();
decltype(foo.x) a = 0;  //a -> int, 类访问表达式
decltype((foo.x)) b = a;//b -> const int &, exp的左值引用
    • exp是函数调用，decltype(exp)和返回值的类型一致
    • 其他情况，若exp是一个左值，则decltype(exp)是exp类型的左值引用，否则和exp类型一致
  • auto和decltype结合使用
      // 复杂冗余的写法
  template <typename R,typename T,typename U>
  R add(T t, U u)
  {
      return t+u;
  } 
  int a = 1; float b = 2.0;
  auto c = add<decltype(a+b)>(a, b);
  // 干净利落的做法
  template <typename T,typename U>
  auto add(T t, U u) -> decltype(t+u)
  {
      return t+u;
  }
  // 另一个例子
  int& foo(int& i);
  float foo(float& f);
  template <typename T>
  auto func(T& val) -> decltype(foo(val))
  {
      return foo(val);
  }

    模板改进

  • 模板别名

      template <typename Val>
  using str_map_t = std::map<std::string, Val>;
  //...
  str_map_t<int> map1;

    using 的别名用法覆盖了typedef的全部功能

      typedef void (*func_t)(int, int);
  //
  using func_t = void(*)(int, int);
  //c++ 98/03
  template <typename T>
  struct func_t
  {
      typedef void (*type)(T, T);
  };
  //using template func_t
  func_t<int>::type xx_1;

  //c++ 11
  template <typename T>
  using func_t = void (*)(T, T);
  //using template func_t
  func_t<int> xx_2;

    using定义模板别名的方法作为定义类型适用于定义阶段

  • 默认参数

      template <typename R = int, typename U>
  R func(U val)
  {
      return val;
  }

  int main(void)
  {
      func(123);
      func<long>(123);
      return 0;
  }

  //外敷模板禁用形参自动推导
  typename<template T>
  struct Foo
  {
      using type = T;
  };
  //使用时禁止了val的自动推导
  typename Foo<int>::type val;

    不能将默认模板参数当做模板参数自动推导的“建议”，因为模板参数自动推导总是根据实参推导来的，当自动推导生效时，默认模板参数会失效

  • 初始化列表

    • 当类型为聚合类型时才适用无初始化列表构造函数的初始化

    • 利用std::initializer_list类模板完成初始化列表构造函数

      class FooVector
{
  std::vector<int> content;
public:
  FooVector(std::initializer_list<int> list)
  {
      for (auto it = list.begin(); it != list.end(); ++it)
          content.push_back(*it);
  }
};

class FooMap
{
  std::map<int, int> content;
  using pair_t = std::map<int, int>::value_type;
public:
  FooMap(std::initializer_list<pair_t> list)
  {
      for (auto it = list.begin(); it != list.end(); ++it)
          content.insert(*it);
  }
};

FooVector foo_1 = { 1, 2, 3, 4, 5 };
FooMap foo_2 = { { 1, 2}, { 3, 4 }, { 5, 6 } };

    std::function和bind绑定器

  • 可调用对象

    • 均可以像函数那样做调用操作
    • 函数指针
    • 具有operator()成员函数的类对象(仿函数)
    • 可被转换为函数指针的类对象
    • 类成员(函数)指针

    • 普通函数不是可调用对象

      void func(void)
{
  // ...
}

struct Foo
{
  void operator()(void)
  {
      // ...
  }
};

struct Bar
{
  using fr_t = void(*)(void);

  static void func(void)
  {
      // ...
  }

  operator fr_t(void)
  {
      return func;
  }
};

struct A
{
  int a;

  void func(void)
  {
      // ...
  }
};

int main(void)
{
  void(* func_ptr)(void) = &func;             //1. 函数指针
  func_ptr();

  Foo foo;                                    //2. 仿函数
  foo;

  Bar bar;                                    //3. 可被转换为函数指针的类对象
  bar();

  void (A::*mem_func_ptr)(void) = &A::func;   //4. 类成员函数指针
  int A::*a_ptr = &A::a;
  A aa;
  (aa.*mem_func_ptr)();                       //   或是类成员指针
  aa.*a_ptr = 123;

  return 0;
}

  • std::function是可调用对象的包装器

    • 基本用法

      #include<iostream>
#include<functional>

void func(void)
{
  std::cout << __Function__ << std::endl;
}

class Foo
{
public:
  static int foo_func(int a)
  {
      std::cout << __Function__ << "(" << a << ") -> :";
      return a;
  }
};

class Bar
{
public:
  int operator()(int a)
  {
      std::cout << __Function__ << "(" << a << ") -> :";
      return a;
  }
};

int main()
{
  std::function<void(void)> fr1 = func;
  fr1();

  std::function<int(int)> fr2 = Foo::foo_func;    // 绑定普通函数
  std::cout<< fr2(123) << std::endl;

  Bar bar;
  fr2 = bar;                                      // 绑定仿函数
  std::cout<< fr2(123) << std::endl;              

  return 0;
}

    • 作为回调函数

      #include<iostream>
#include<functional>

class A
{
  std::function<void()> callback;
public:
  A(const std::function<void()>& f) : callback(f) { }

  void notify(void)
  {
      callback();
  }
};

class Foo
{
public:
  void operator()(void)
  {
      std::cout << __Func__ << std::endl;
  }
};

int main()
{
  Foo foo;
  A aa(foo);
  // do something...
  aa.notify;

  return 0;
}

    • 作为函数入参

      #include<iostream>
#include<functional>

void call_when_even(int x, const std::function<void(int)>& f)
{
  if (!(x & 1))   // x % 2 == 0
  {
      f(x);
  }
}

void output(int x)
{
  std::cout << x << " ";
}

int main(void)
{
  for(int i = 0; i < 10; ++i)
  {
      call_when_even(i, output);
  }

  return 0;
}
    • 当给std::function填入合适的函数签名(即函数类型，包括返回值类型与参数表)之后，它就变成了一个可以容纳所有这一类调用方式的”函数包装器”
    • std::function比普通函数指针更加灵活，可以取代函数指针的作用，可以用来保存函数延迟执行，适合作为回调函数

  • std::bind绑定器

    • std::bind 用来将可调用对象与其参数一起绑定。绑定后的结果可以使用 std::function 进行保存，并延迟调用
    • 将可调用对象与其参数一起绑定一起绑定成一个仿函数

    • 只绑定部分参数，将多元可调用对象转成一元或多元可调用对象

      #include<iostream>
#include<functional>

void call_when_even(int x, const std::function<void(int)>& f)
{
  if (!(x & 1))
  {
      f(x);
  }
}

void output(int x)
{
  std::cout << x << " ";
}

void output_add_2(int x)
{
  std::cout << x+2 << " ";
}

int main(void)
{
  {
      auto fr = std::bind(output, std::placeholders::_1);
      for (int i = 0; i < 10; ++i)
      {
          call_when_even(i, fr);
      }
      std::cout << std::endl;
  }
  {
      auto fr = std::bind(output_add_2, std::placeholders::_1);
      for (int i = 0; i < 10; ++i)
      {
          call_when_even(i, fr);
      }
      std::cout << std::endl;
  }
return 0;
}
    • std::placeholders::_1是一个占位符，代表这个位置将在函数调用时，被传入的第一个参数所替代。

    • std::bind 的返回类型是一个仿函数类型

      #include<iostream>
#include<functional>

void output(int x, int y)
{
  std::cout << x << " " << y << endl;
}

int main()
{
  std::bind(output, 1, 2)(); // 1 2
  std::bind(output, std::placeholders::_1, std::placeholders::_2)(1, 2); // 2 1
  return 0;
}

    • 在绑定部分参数时，通过使用std::placeholders来决定空位参数将会调用属于调用发生时的第几个参数

      #include<iostream>
#include<functional>

class A
{
public:
  int i = 0;

  void output(int x)
  {

  }
};

int main()
{
  A a;
  std::function<void(int)> fr = std::bind(&A::output, &a, std::placeholders::_1, std::placeholders::_2);
  fr(1, 2);

  std::function<int&(void)> fr_i = std::bind(&A::i, &a);
  fr_i() = 123;
  std::cout << a.i << std::endl;

  return 0;
}
    • 利用std::bind 构建功能闭包

    • 闭包就是一种语法糖，它以很自然的形式，把我们的目的和我们的目的所涉及的资源全给自动打包在一起，以某种自然、尽量不让人误解的方式让人来使用。

      std::bind(std::greater<int>(), std::placeholders::_1, 5);
std::bind(std::less_equal<int>(), std::placeholders::_1, 10);

using std::placeholders::_1;
auto f = std::bind(std::logical_and<bool>(), std::bind(std::greater<int>(), _1, 5), std::bind(std::less_equal<int>(), _1, 10));
int count  = std::count_if(coll.begin(), coll.end(), f); //查找集合中大于5小于10的元素个数

  • lambda表达式

    • 可以认为lambda表达式是一个仿函数类型，所以可以用std::functon与std::bind 来储存和操作lambda表达式

    • 对于没有捕获任何变量的lambda表达式，还可以被转换为一个普通的函数指针

      std::function<int(int)> f1 = [](int a){ return a; };
std::function<int(void)> f2 = std::bind([](int a){ return a; }, 123);

using func_t = int(*)(int);
func_t f = [](int a){ return a; };
f(123);
    • 可以就地实现闭包
      int count = std::count_if(coll.begin(), coll.end(), [](int x){return x > 5 && x < 10});

  • tuple元组

    • tuple元组是一个固定大小的不同类型值的集合，是泛化的std::pair

      //构建元组
tuple<const char*, int> tp = make_tuple(sendPack, nSendSize);

//利用std::tie 创建元组的左值引用
auto tp = return std::tie(1, "aa", 2);
//tp的类型是 std::tuple<int&, string&, int&>

//创建右值的引用元组
std::map<int, std::string> m;
m.emplace(std::forward_as_tuple(10, std::string(20, 'a'));

/* -----------------------------------*/
//获取元组的值
const char* data = tp.get<0>();
int len = tp.get<1>();

int x,y;
string a;
std::tie(x,a,y) = tp;
//利用占位符表示不解某个位置的值
std::tie(std::ignore,std::ignore, y) = tp;

/*------------------------------------*/
//利用tuple_cat连接多个tuple
int main()
{
  std::tuple<int, std::string, float> t1(10, "Test", 3.14);
  int n = 7;
  auto t2 = std::tuple_cat(t1, std::make_pair("Foo", "bar"), t1, std::tie(n));
  n = 10;
  print(t2);
  return 0;
}
// 输出 10,Test,3.14,Foo,bar,10,Test,3.14,10
    • tuple的优点是能容纳任意类型和任意数量的元素
